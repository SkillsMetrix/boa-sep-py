
class User:
    name='shashi'
    state='MH'

    # constructor
    def __init__(self,email="user@mail.com",city="pune"):

        self.email=email
        self.city=city

    # regular method
    def printData(self):
        print(self.name,self.state)

# creating object
userdata= User(email="admin@mail.com",city="Chennai")
print(userdata.city,userdata.email)
userdata.printData()
